 36:         $pks = array_unique($pks);
 37:         JArrayHelper::toInteger($pks);
 38: 
 39:         // Remove any values of zero.
 40:         if (array_search(0, $pks, true)) {
 41:             unset($pks[array_search(0, $pks, true)]);
 42:         }
 43: 
 44:         if (empty($pks)) {
 45:             $this->setError(JText::_('JGLOBAL_NO_ITEM_SELECTED'));
 46:             return false;
 47:         }
 48: 
 49:         $done = false;
 50: 
 51:         if (!empty($commands['category_id'])) {
 52:             $cmd = JArrayHelper::getValue($commands, 'move_copy', 'c');
 53: 
 54:             if ($cmd == 'c') {
 55:                 $result = $this->batchCopy($commands['category_id'], $pks, $contexts);
 56:                 if (is_array($result)) {
 57:                     $pks = $result;
 58:                 } else {
 59:                     return false;
 60:                 }
 61:             } elseif ($cmd == 'm' && !$this->batchMove($commands['category_id'], $pks, $contexts)) {
 62:                 return false;
 63:             }
 64:             $done = true;
 65:         }
 66: 
 67:         if (!empty($commands['assetgroup_id'])) {
 68:             if (!$this->batchAccess($commands['assetgroup_id'], $pks, $contexts)) {
 69:                 return false;
 70:             }
 71: 
 72:             $done = true;
 73:         }
 74: 
 75:         if (!empty($commands['language_id'])) {
 76:             if (!$this->batchLanguage($commands['language_id'], $pks, $contexts)) {
 77:                 return false;
 78:             }
 79: 
 80:             $done = true;
 81:         }
 82: 
 83:         if (strlen($commands['user_id']) > 0) {
 84:             if (!$this->batchUser($commands['user_id'], $pks, $contexts)) {
 85:                 return false;
 86:             }
 87: 
 88:             $done = true;
 89:         }
 90: 
 91:         if (!$done) {
 92:             $this->setError(JText::_('JLIB_APPLICATION_ERROR_INSUFFICIENT_BATCH_INFORMATION'));
 93:             return false;
 94:         }
 95: 
 96:         // Clear the cache
 97:         $this->cleanCache();
