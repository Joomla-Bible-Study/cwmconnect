 11:  * A smart INI file parser with reproducible behaviour among different PHP versions
 12:  */
 13: class LiveUpdateINIHelper
 14: {
 15:     /**
 16:      * Parse an INI file and return an associative array. Since PHP versions before
 17:      * 5.1 are bitches with regards to INI parsing, I use a PHP-only solution to
 18:      * overcome this obstacle.
 19:      * @param   string  $file   The file to process
 20:      * @param   bool    $process_sections   True to also process INI sections
 21:      * @return  array   An associative array of sections, keys and values
 22:      */
 23:     public static function parse_ini_file( $file, $process_sections, $rawdata = false )
 24:     {
 25:         if($rawdata)
 26:         {
 27:             return self::parse_ini_file_php($file, $process_sections, $rawdata);
 28:         }
 29:         else
 30:         {
 31:             if( version_compare(PHP_VERSION, '5.1.0', '>=') && (!$rawdata) )
 32:             {
 33:                 if( function_exists('parse_ini_file') )
 34:                 {
 35:                     return parse_ini_file($file, $process_sections);
 36:                 }
 37:                 else
 38:                 {
 39:                     return self::parse_ini_file_php($file, $process_sections);
 40:                 }
 41:             } else {
 42:                 return self::parse_ini_file_php($file, $process_sections, $rawdata);
 43:             }
 44:         }
 45:     }
 46: 
 47:     /**
 48:      * A PHP based INI file parser.
 49:      * Thanks to asohn ~at~ aircanopy ~dot~ net for posting this handy function on
 50:      * the parse_ini_file page on http://gr.php.net/parse_ini_file
 51:      * @param   string  $file   Filename to process
 52:      * @param   bool    $process_sections   True to also process INI sections
 53:      * @param   bool    $rawdata    If true, the $file contains raw INI data, not a filename
 54:      * @return  array   An associative array of sections, keys and values
 55:      */
 56:     static function parse_ini_file_php($file, $process_sections = false, $rawdata = false)
 57:     {
 58:         $process_sections = ($process_sections !== true) ? false : true;
 59: 
 60:         if(!$rawdata)
 61:         {
 62:             $ini = file($file);
 63:         }
 64:         else
 65:         {
 66:             $file = str_replace("\r","",$file);
 67:             $ini = explode("\n", $file);
 68:         }
 69: 
 70:         if (count($ini) == 0) {return array();}
 71: 
 72:         $sections = array();
 73:         $values = array();
 74:         $result = array();
 75:         $globals = array();
 76:         $i = 0;
 77:         foreach ($ini as $line) {
 78:             $line = trim($line);
 79:             $line = str_replace("\t", " ", $line);
 80: 
 81:             // Comments
 82:             if (!preg_match('/^[a-zA-Z0-9[]/', $line)) {continue;}
 83: 
 84:             // Sections
 85:             if ($line{0} == '[') {
 86:                 $tmp = explode(']', $line);
 87:                 $sections[] = trim(substr($tmp[0], 1));
 88:                 $i++;
 89:                 continue;
 90:             }
 91: 
 92:             // Key-value pair
 93:             list($key, $value) = explode('=', $line, 2);
 94:             $key = trim($key);
 95:             $value = trim($value);
 96:             if (strstr($value, ";")) {
 97:                 $tmp = explode(';', $value);
 98:                 if (count($tmp) == 2) {
 99:                     if ((($value{0} != '"') && ($value{0} != "'")) ||
100:                     preg_match('/^".*"\s*;/', $value) || preg_match('/^".*;[^"]*$/', $value) ||
101:                     preg_match("/^'.*'\s*;/", $value) || preg_match("/^'.*;[^']*$/", $value) ){
102:                         $value = $tmp[0];
103:                     }
104:                 } else {
105:                     if ($value{0} == '"') {
106:                         $value = preg_replace('/^"(.*)".*/', '$1', $value);
107:                     } elseif ($value{0} == "'") {
108:                         $value = preg_replace("/^'(.*)'.*/", '$1', $value);
109:                     } else {
110:                         $value = $tmp[0];
111:                     }
112:                 }
113:             }
114:             $value = trim($value);
115:             $value = trim($value, "'\"");
116: 
117:             if ($i == 0) {
118:                 if (substr($line, -1, 2) == '[]') {
119:                     $globals[$key][] = $value;
120:                 } else {
121:                     $globals[$key] = $value;
122:                 }
123:             } else {
124:                 if (substr($line, -1, 2) == '[]') {
125:                     $values[$i-1][$key][] = $value;
126:                 } else {
127:                     $values[$i-1][$key] = $value;
128:                 }
129:             }
130:         }
131: 
132:         for($j = 0; $j < $i; $j++) {
133:             if ($process_sections === true) {
134:                 if( isset($sections[$j]) && isset($values[$j]) )    $result[$sections[$j]] = $values[$j];
135:             } else {
136:                 if( isset($values[$j]) ) $result[] = $values[$j];
137:             }
138:         }
139: 
140:         return $result + $globals;
141:     }
142: }
