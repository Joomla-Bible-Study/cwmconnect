365:     /**
366:      * Method to get the SQL query used to retrieve the list of content items.
367:      *
368:      * @param   mixed  $sql  A JDatabaseQuery object or null.
369:      *
370:      * @return  JDatabaseQuery  A database object.
371:      *
372:      * @since   1.7.0
373:      */
374:     protected function getListQuery($sql = null) {
375:         $db = JFactory::getDbo();
376:         // Check if we can use the supplied SQL query.
377:         $sql = is_a($sql, 'JDatabaseQuery') ? $sql : $db->getQuery(true);
378:         $sql->select('a.id, a.name AS title, a.alias, a.address, a.created AS start_date');
379:         $sql->select('a.created_by_alias, a.modified, a.modified_by');
380:         $sql->select('a.metakey, a.metadesc, a.metadata, a.language');
381:         $sql->select('a.sortname1, a.sortname2, a.sortname3');
382:         $sql->select('a.publish_up AS publish_start_date, a.publish_down AS publish_end_date');
383:         $sql->select('a.suburb AS city, a.state AS region, a.country, a.postcode AS zip');
384:         $sql->select('a.telephone, a.fax, a.misc AS summary, a.email_to AS email, a.mobile');
385:         $sql->select('a.webpage, a.access, a.published AS state, a.ordering, a.params, a.catid');
386:         $sql->select('c.title AS category, c.published AS cat_state, c.access AS cat_access');
387: 
388:         // Handle the alias CASE WHEN portion of the query
389:         $case_when_item_alias = ' CASE WHEN ';
390:         $case_when_item_alias .= $sql->charLength('a.alias');
391:         $case_when_item_alias .= ' THEN ';
392:         $a_id = $sql->castAsChar('a.id');
393:         $case_when_item_alias .= $sql->concatenate(array($a_id, 'a.alias'), ':');
394:         $case_when_item_alias .= ' ELSE ';
395:         $case_when_item_alias .= $a_id . ' END as slug';
396:         $sql->select($case_when_item_alias);
397: 
398:         $case_when_category_alias = ' CASE WHEN ';
399:         $case_when_category_alias .= $sql->charLength('c.alias');
400:         $case_when_category_alias .= ' THEN ';
401:         $c_id = $sql->castAsChar('c.id');
402:         $case_when_category_alias .= $sql->concatenate(array($c_id, 'c.alias'), ':');
403:         $case_when_category_alias .= ' ELSE ';
404:         $case_when_category_alias .= $c_id . ' END as catslug';
405:         $sql->select($case_when_category_alias);
406: 
407:         $sql->select('u.name AS user');
408:         $sql->from('#__churchdirectory_details AS a');
409:         $sql->join('LEFT', '#__categories AS c ON c.id = a.catid');
410:         $sql->join('LEFT', '#__users AS u ON u.id = a.user_id');
411: 
412:         return $sql;
413:     }
