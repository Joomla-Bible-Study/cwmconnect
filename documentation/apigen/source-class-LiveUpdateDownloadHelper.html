222:     private static function &getFOPEN($url, $fp = null)
223:     {
224:         $result = false;
225:         
226:         // Track errors
227:         if( function_exists('ini_set') ) {
228:             $track_errors = ini_set('track_errors',true);
229:         }
230:         
231:         // Open the URL for reading
232:         if(function_exists('stream_context_create')) {
233:             // PHP 5+ way (best)
234:             $httpopts = Array('user_agent'=>'Mozilla/4.0 (compatible; MSIE 7.0; Windows NT 5.1; .NET CLR 1.0.3705; .NET CLR 1.1.4322; Media Center PC 4.0)');
235:             $context = stream_context_create( array( 'http' => $httpopts ) );
236:             $ih = @fopen($url, 'r', false, $context);
237:         } else {
238:             // PHP 4 way (actually, it's just a fallback as we can't run Admin Tools in PHP4)
239:             if( function_exists('ini_set') ) {
240:                 ini_set('user_agent', 'Mozilla/4.0 (compatible; MSIE 7.0; Windows NT 5.1; .NET CLR 1.0.3705; .NET CLR 1.1.4322; Media Center PC 4.0)');
241:             }
242:             $ih = @fopen($url, 'r');
243:         }
244:         
245:         // If fopen() fails, abort
246:         if( !is_resource($ih) ) {
247:             return $result;
248:         }
249:         
250:         // Try to download
251:         $bytes = 0;
252:         $result = true;
253:         $return = '';
254:         while (!feof($ih) && $result)
255:         {
256:             $contents = fread($ih, 4096);
257:             if ($contents === false) {
258:                 @fclose($ih);
259:                 $result = false;
260:                 return $result;
261:             } else {
262:                 $bytes += strlen($contents);
263:                 if(is_resource($fp)) {
264:                     $result = @fwrite($fp, $contents);
265:                 } else {
266:                     $return .= $contents;
267:                     unset($contents);
268:                 }
269:             }
270:         }
271:         
272:         @fclose($ih);
273:         
274:         if(is_resource($fp)) {
275:             return $result;
276:         } elseif( $result === true ) {
277:             return $return;
278:         } else {
279:             return $result;
280:         }
281:     }
