 10: /**
 11:  * Live Update Component Storage Class
 12:  * Allows to store the update data to a component's parameters. This is the most reliable method. 
 13:  * Its configuration options are:
 14:  * component    string  The name of the component which will store our data. If not specified the extension name will be used.
 15:  * key          string  The name of the component parameter where the serialized data will be stored. If not specified "liveupdate" will be used.
 16:  */
 17: class LiveUpdateStorageComponent extends LiveUpdateStorage
 18: {
 19:     private static $component = null;
 20:     private static $key = null;
 21:     
 22:     public function load($config)
 23:     {
 24:         if(!array_key_exists('component', $config)) {
 25:             self::$component = $config['extensionName'];
 26:         } else {
 27:             self::$component = $config['component'];
 28:         }
 29: 
 30:         if(!array_key_exists('key', $config)) {
 31:             self::$key = 'liveupdate';
 32:         } else {
 33:             self::$key = $config['key'];
 34:         }
 35:         
 36:         jimport('joomla.html.parameter');
 37:         jimport('joomla.application.component.helper');
 38:         // Not using JComponentHelper to avoid conflicts ;)
 39:         $db = JFactory::getDbo();
 40:         if( version_compare(JVERSION,'1.6.0','ge') ) {
 41:             $sql = $db->getQuery(true)
 42:                 ->select($db->qn('params'))
 43:                 ->from($db->qn('#__extensions'))
 44:                 ->where($db->qn('type').' = '.$db->q('component'))
 45:                 ->where($db->qn('element').' = '.$db->q(self::$component));
 46:         } else {
 47:             $sql = 'SELECT '.$db->nameQuote('params').' FROM '.$db->nameQuote('#__components').
 48:                 ' WHERE '.$db->nameQuote('option').' = '.$db->Quote(self::$component).
 49:                 " AND `parent` = 0 AND `menuid` = 0";
 50:         }
 51:         $db->setQuery($sql);
 52:         $rawparams = $db->loadResult();
 53:         if(version_compare(JVERSION, '1.6.0', 'ge')) {
 54:             $params = new JRegistry();
 55:             $params->loadJSON($rawparams);
 56:         } else {
 57:             $params = new JParameter($rawparams);
 58:         }
 59:         $data = $params->getValue(self::$key, '');
 60:                 
 61:         jimport('joomla.registry.registry');
 62:         self::$registry = new JRegistry('update');
 63:         
 64:         self::$registry->loadINI($data);
 65:     }
 66:     
 67:     public function save()
 68:     {
 69:         $data = self::$registry->toString('INI');
 70:         
 71:         $db = JFactory::getDBO();
 72:         
 73:         // An interesting discovery: if your component is manually updating its
 74:         // component parameters before Live Update is called, then calling Live
 75:         // Update will reset the modified component parameters because
 76:         // JComponentHelper::getComponent() returns the old, cached version of
 77:         // them. So, we have to forget the following code and shoot ourselves in
 78:         // the feet. Dammit!!!
 79:         /*
 80:         jimport('joomla.html.parameter');
 81:         jimport('joomla.application.component.helper');
 82:         $component = JComponentHelper::getComponent(self::$component);
 83:         $params = new JParameter($component->params);
 84:         $params->setValue(self::$key, $data);
 85:         */
 86: 
 87:         if( version_compare(JVERSION,'1.6.0','ge') ) {
 88:             $sql = $db->getQuery(true)
 89:                 ->select($db->qn('params'))
 90:                 ->from($db->qn('#__extensions'))
 91:                 ->where($db->qn('type').' = '.$db->q('component'))
 92:                 ->where($db->qn('element').' = '.$db->q(self::$component));
 93:         } else {
 94:             $sql = 'SELECT '.$db->nameQuote('params').' FROM '.$db->nameQuote('#__components').
 95:                 ' WHERE '.$db->nameQuote('option').' = '.$db->Quote(self::$component).
 96:                 " AND `parent` = 0 AND `menuid` = 0";
 97:         }
 98:         $db->setQuery($sql);
 99:         $rawparams = $db->loadResult();
100:         $params = new JRegistry();
101:         if( version_compare(JVERSION,'1.6.0','ge') ) {
102:             $params->loadJSON($rawparams);
103:         } else {
104:             $params->loadINI($rawparams);
105:         }
106:         
107:         $params->setValue(self::$key, $data);
108:         
109:         if( version_compare(JVERSION,'1.6.0','ge') )
110:         {
111:             // Joomla! 1.6
112:             $data = $params->toString('JSON');
113:             $sql = $db->getQuery(true)
114:                 ->update($db->qn('#__extensions'))
115:                 ->set($db->qn('params').' = '.$db->q($data))
116:                 ->where($db->qn('type').' = '.$db->q('component'))
117:                 ->where($db->qn('element').' = '.$db->q(self::$component));
118:         }
119:         else
120:         {
121:             // Joomla! 1.5
122:             $data = $params->toString('INI');
123:             $sql = 'UPDATE `#__components` SET `params` = '.$db->Quote($data).' WHERE '.
124:                 "`option` = ".$db->Quote(self::$component)." AND `parent` = 0 AND `menuid` = 0";
125:         }
