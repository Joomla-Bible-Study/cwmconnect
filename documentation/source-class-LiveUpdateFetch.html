 34:         $config = LiveUpdateConfig::getInstance();
 35:         $extInfo = $config->getExtensionInformation();

 37:         // Filter by stability level
 38:         $minStability = $config->getMinimumStability();
 39:         $stability = strtolower($updateInfo->stability);
 40:         
 41:         switch($minStability) {
 42:             case 'alpha':

 44:                 // Reports any stability level as an available update
 45:                 break;
 46:             
 47:             case 'beta':
 48:                 // Do not report alphas as available updates
 49:                 if(in_array($stability, array('alpha'))) return 0;
 50:                 break;
 51:             
 52:             case 'rc':
 53:                 // Do not report alphas and betas as available updates
 54:                 if(in_array($stability, array('alpha','beta'))) return 0;
 55:                 break;
 56:                 
 57:             case 'stable':
 58:                 // Do not report alphas, betas and rcs as available updates
 59:                 if(in_array($stability, array('alpha','beta','rc'))) return 0;
 60:                 break;
 61:         }
 62:         
 63:         // Use the version strategy to determine the availability of an update
 64:         switch($config->getVersionStrategy()) {
 65:             case 'newest':
 66:                 jimport('joomla.utilities.date');
 67:                 if(empty($extInfo)) {
 68:                     $mine = new JDate('2000-01-01 00:00:00');
 69:                 } else {
 70:                     $mine = new JDate($extInfo['date']);
 71:                 }
 72:                 
 73:                 $theirs = new JDate($updateInfo->date);
 74:                 
 75:                 return ($theirs->toUnix() > $mine->toUnix()) ? 1 : 0;
 76:                 break;
 77:             
 78:             case 'vcompare':
 79:                 $mine = $extInfo['version'];
 80:                 if(empty($mine)) $mine = '0.0.0';
 81:                 $theirs = $updateInfo->version;
 82:                 if(empty($theirs)) $theirs = '0.0.0';
 83:                 
 84:                 return (version_compare($theirs, $mine, 'gt')) ? 1 : 0;
 85:                 break;
 86:             
 87:             case 'different':
 88:                 $mine = $extInfo['version'];
 89:                 if(empty($mine)) $mine = '0.0.0';
 90:                 $theirs = $updateInfo->version;
 91:                 if(empty($theirs)) $theirs = '0.0.0';
 92:                 
 93:                 return ($theirs != $mine) ? 1 : 0;
 94:                 break;
 95:         }

 98:     /**
 99:      * Get the latest version (update) information, either from the cache or
100:      * from the update server.
101:      * 
102:      * @param $force bool Set to true to force fetching fresh data from the server
103:      * 
104:      * @return stdClass The update information, in object format
105:      */
106:     public function getUpdateInformation($force = false)
107:     {
108:         // Get the Live Update configuration
109:         $config = LiveUpdateConfig::getInstance();
110:         
111:         // Get an instance of the storage class
112:         $storageOptions = $config->getStorageAdapterPreferences();
113:         require_once dirname(__FILE__).'/storage/storage.php';
114:         $this->storage = LiveUpdateStorage::getInstance($storageOptions['adapter'], $storageOptions['config']);
115:         $storage = $this->storage;
116:         
117:         // Fetch information from the cache
118:         if(version_compare(JVERSION, '1.6.0', 'ge')) {
119:             $registry = $storage->getRegistry();
120:             $lastCheck = $registry->get('lastcheck', 0);
121:             $cachedData = $registry->get('updatedata', null);
122:         } else {
123:             $lastCheck = $storage->get('lastcheck', 0);
124:             $cachedData = $storage->get('updatedata', null);
125:         }
126:         
127:         if(is_string($cachedData)) {
128:             $cachedData = trim($cachedData,'"');
129:             $cachedData = json_decode($cachedData);
130:         }
131:         
132:         if(empty($cachedData)) {
133:             $lastCheck = 0;
134:         }
135:         
136:         // Check if the cache is at most $cacheTTL hours old
137:         $now = time();
138:         $maxDifference = $this->cacheTTL * 3600;
139:         $difference = abs($now - $lastCheck);
140: 
141:         if(!($force) && ($difference <= $maxDifference)) {
142:             // The cache is fresh enough; return cached data
143:             return $cachedData;
144:         } else {
145:             // The cache is stale; fetch new data, cache it and return it to the caller
146:             $data = $this->getUpdateData($force);
147:             $this->storage->set('lastcheck', $now);
148:             $this->storage->set('updatedata', json_encode($data));
149:             $this->storage->save();
150:             return $data;
151:         }
152:     }

256:     private function parseINI($rawData)

258:         $ret = array(

268:         $magicPos = strpos($rawData, '; Live Update provision file');
269:         
270:         if($magicPos === false) {
271:             // That's not an INI file :(
272:             return $ret;
273:         }
274:         
275:         if($magicPos !== 0) {
276:             $rawData = substr($rawData, $magicPos);
277:         }
278:         
279:         require_once dirname(__FILE__).'/inihelper.php';
280:         $iniData = LiveUpdateINIHelper::parse_ini_file($rawData, false, true);
281:         
282:         $ret['version'] = $iniData['version'];
283:         $ret['date'] = $iniData['date'];
284:         $config = LiveUpdateConfig::getInstance();
285:         $auth = $config->getAuthorization();
286:         $glue = strpos($iniData['link'],'?') === false ? '?' : '&';
287:         $ret['downloadURL'] = $iniData['link'] . (empty($auth) ? '' : $glue.$auth);
288:         if(array_key_exists('stability', $iniData)) {
289:             $stability = $iniData['stability'];
290:         } else {
291:             // Stability not defined; guesswork mode enabled
292:             $version = $ret['version'];
293:             if( preg_match('#^[0-9\.]*a[0-9\.]*#', $version) == 1 ) {
294:                 $stability = 'alpha';
295:             } elseif( preg_match('#^[0-9\.]*b[0-9\.]*#', $version) == 1 ) {
296:                 $stability = 'beta';
297:             } elseif( preg_match('#^[0-9\.]*rc[0-9\.]*#', $version) == 1 ) {
298:                 $stability = 'rc';
299:             } elseif( preg_match('#^[0-9\.]*$#', $version) == 1 ) {
300:                 $stability = 'stable';
301:             } else {
302:                 $stability = 'svn';
303:             }
304:         }
305:         $ret['stability'] = $stability;
306:         
307:         if(array_key_exists('releasenotes', $iniData)) {
308:             $ret['releasenotes'] = $iniData['releasenotes'];
309:         }
310:         
311:         if(array_key_exists('infourl', $iniData)) {
312:             $ret['infoURL'] = $iniData['infourl'];
313:         }
314: 
315:         return $ret;
